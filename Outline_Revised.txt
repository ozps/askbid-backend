    -password: nvarchar

    -tel_no: nvarchar

    -bank_no: nvarchar

    -id: int
    -brand: nvarchar
    -desc: nvarchar
    -color: nvarchar
    -released_date: date
    -image: nvarchar
    -visited_count: int

•order => 1 order per 1 item
    -id: int
    -user_id: int 
    -item_id: int 
    -size: nvarchar 
    -price: double 
    -type: (Ask:0,Bid:1)

    -id: int

    -order_id: [int] 
    -amount: int
    -total_price: double
    -paid_date: date
    -receive: (No:0, Yes:1)
    -stamp_date: date
