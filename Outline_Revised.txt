Table
•user
    -id: int
    -full_name: nvarchar
    -card_no: nvarchar  
    -email: nvarchar
    -password: nvarchar
    -address: nvarchar
    -tel_no: nvarchar
    -balance: float
    -bank_no: nvarchar
    -avatar_image: nvarchar
    -card_image: nvarchar
    -level: (ban:-1,normal:0,verified:1,admin:2)

•item 
    -id: int
    -brand: nvarchar
    -desc: nvarchar
    -color: nvarchar
    -image: nvarchar
    -released_date: date
    -visited_count: int

•order => 1 order per 1 item
    -id: int
    -user_id: int 
    -item_id: int 
    -size: nvarchar 
    -price: float 
    -type: (Ask:0,Bid:1)
    -published_date: date
    -out_of_stock: (No:0,Yes:1) 

•match
    -id: int
    -user_id: int 
    -order_id: [int] 
    -amount: int
    -total_price: float 
    -paid_date: date
    -receive: (No:0, Yes:1)
    -stamp_date: date

-upload_avatar      POST in-user_id

•/admin (verified=2)
-get_all_cards 		POST

-add_item 		 	POST in-`item` (permanent)

•/items
-get_all_items		GET
-get_new_items		GET
-get_popular_items	GET

•/orders

-get_user_order 	POST in-user_id
-get_order 			POST in-order_id
-get_ask_price 		GET in-item_id
-get_bid_price 		GET in-item_id
